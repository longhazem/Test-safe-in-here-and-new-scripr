-- Bu döngü her 3 saniyede bir ağır tarama yapar, FPS düşüşünü engeller.
task.spawn(function()
    while true do
        if ESPState.NPCEnabled or ESPState.MineEnabled then
            local newNPCs = {}
            local newMines = {}
            
            for _, obj in ipairs(Workspace:GetDescendants()) do
                -- NPC Taraması
                if ESPState.NPCEnabled and obj:IsA("Model") and obj.Name == "Male" and not Players:GetPlayerFromCharacter(obj) then
                    table.insert(newNPCs, obj)
                end
                -- Mine Taraması
                if ESPState.MineEnabled and (obj.Name:match("Mines") or obj.Name:match("Mine")) then
                    for _, m in ipairs(obj:GetDescendants()) do
                        if m:IsA("BasePart") then table.insert(newMines, m) end
                    end
                end
            end
            ESPState.CachedNPCs = newNPCs
            ESPState.CachedMines = newMines
        end
        task.wait(3) 
    end
end)

local function IsVisible(targetPart)
    local rayOrigin = Camera.CFrame.Position
    local rayDirection = targetPart.Position - rayOrigin
    local raycastParams = RaycastParams.new()
    raycastParams.FilterDescendantsInstances = {Camera, LocalPlayer.Character}
    raycastParams.FilterType = Enum.RaycastFilterType.Exclude
    local raycastResult = Workspace:Raycast(rayOrigin, rayDirection, raycastParams)
    return raycastResult and raycastResult.Instance:IsDescendantOf(targetPart.Parent)
end

local FOVCircle = Drawing.new("Circle")
FOVCircle.Thickness = 1
FOVCircle.Color = Color3.fromRGB(255, 0, 0)

local function GetClosestTarget()
    local closestPart = nil
    local shortestDist = AimbotState.FOV
    local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
    for _, obj in ipairs(ESPState.CachedNPCs) do
        local targetPart = obj:FindFirstChild(AimbotState.TargetPart)
        if targetPart then
            local pos, onScreen = Camera:WorldToViewportPoint(targetPart.Position)
            if onScreen then
                local dist = (Vector2.new(pos.X, pos.Y) - center).Magnitude
                if dist <= shortestDist then
                    closestPart = targetPart
                    shortestDist = dist
                end
            end
        end
    end
    return closestPart
end

RunService.RenderStepped:Connect(function()
    -- Fullbright
    if FullbrightEnabled then
        Lighting.Ambient = Color3.fromRGB(255, 255, 255)
        Lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
        Lighting.Brightness = 2
    end

    -- NPC ESP Render
    if ESPState.NPCEnabled then
        for _, obj in ipairs(ESPState.CachedNPCs) do
            local root = obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChild("Head")
            if root then
                local hl = obj:FindFirstChild("NPCESP_XR") or Instance.new("Highlight", obj)
                hl.Name = "NPCESP_XR"
                hl.FillTransparency = 1
                hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
                hl.OutlineColor = IsVisible(root) and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)
                hl.OutlineTransparency = ESPState.NPCTransparency
                
                local pos, onScreen = Camera:WorldToViewportPoint(root.Position)
                if onScreen then
                    if not ESPState.Drawings[obj] then
                        local t = Drawing.new("Text") t.Size = 14 t.Center = true t.Outline = true t.Color = Color3.fromRGB(255, 255, 255)
                        ESPState.Drawings[obj] = t
                    end
                    local d = ESPState.Drawings[obj]
                    d.Position = Vector2.new(pos.X, pos.Y + 15)
                    d.Text = string.format("[%d m]", (Camera.CFrame.Position - root.Position).Magnitude)
                    d.Visible = false
                elseif ESPState.Drawings[obj] then
                    ESPState.Drawings[obj].Visible = false
                end
            end
        end
    end

    -- Mine ESP Render
    if ESPState.MineEnabled then
        for _, mine in ipairs(ESPState.CachedMines) do
            local hl = mine:FindFirstChild("MineESP_XR") or Instance.new("Highlight", mine)
            hl.Name = "MineESP_XR"
            hl.FillTransparency = 1
            hl.OutlineColor = Color3.fromRGB(0, 150, 255)
            hl.OutlineTransparency = ESPState.MineTransparency
            hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        end
    end

    -- Aimbot Logic
    if AimbotState.Enabled then
        local center = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
        FOVCircle.Position = center
        FOVCircle.Radius = AimbotState.FOV
        if AimbotState.Locking then
            local t = GetClosestTarget()
            if t then
                local p = Camera:WorldToViewportPoint(t.Position)
                mousemoverel((p.X - center.X) * (AimbotState.Smoothness / 4), (p.Y - center.Y) * (AimbotState.Smoothness / 4))
            end
        end
    end
end)
